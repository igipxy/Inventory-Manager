
CREATE DATABASE InventoryManager;
USE InventoryManager;

-- Categories Table
CREATE TABLE Categories (
    category_id INT AUTO_INCREMENT,
    category_name VARCHAR(50) NOT NULL,
    description TEXT NULL,
    PRIMARY KEY (category_id),
    UNIQUE (category_name)
);

-- Supplier Table
CREATE TABLE Suppliers (
    supplier_id INT AUTO_INCREMENT,
    company_name VARCHAR(100) NOT NULL,
    contact_email VARCHAR(100) NULL,
    contact_phone VARCHAR(20) NULL,
    PRIMARY KEY (supplier_id),
    UNIQUE (contact_email)
);

-- Products Table
CREATE TABLE Products (
    product_id INT AUTO_INCREMENT,
    sku VARCHAR(50) NOT NULL,
    product_name VARCHAR(100) NOT NULL,
    category_id INT NULL,
    supplier_id INT NULL,
    unit_price DECIMAL(10,2) NOT NULL,
    reorder_level INT DEFAULT 0,
    PRIMARY KEY (product_id),
    UNIQUE (sku),
    CONSTRAINT fk_product_category 
        FOREIGN KEY (category_id) REFERENCES Categories(category_id) 
        ON DELETE SET NULL 
        ON UPDATE CASCADE,
    CONSTRAINT fk_product_supplier 
        FOREIGN KEY (supplier_id) REFERENCES Suppliers(supplier_id) 
        ON DELETE CASCADE  
        ON UPDATE CASCADE
);

ALTER TABLE Products
ADD COLUMN is_active BOOLEAN DEFAULT TRUE AFTER reorder_level;


-- Inventory_Transactions Table
CREATE TABLE Inventory_Transactions (
    transaction_id INT AUTO_INCREMENT,
    product_id INT NOT NULL,
    transaction_type ENUM('IN', 'OUT', 'ADJUST') NOT NULL,
    quantity INT NOT NULL,
    transaction_date DATETIME DEFAULT CURRENT_TIMESTAMP,
    notes VARCHAR(255) NULL,
    PRIMARY KEY (transaction_id),
    -- If a product is deleted, cascade the deletion to its transaction history
    CONSTRAINT fk_transaction_product 
        FOREIGN KEY (product_id) REFERENCES Products(product_id) 
        ON DELETE CASCADE 
        ON UPDATE CASCADE
);

-- Create the Users table
CREATE TABLE Users (
    id INT AUTO_INCREMENT,
    username VARCHAR(50) NOT NULL,
    email VARCHAR(100) NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id),
    UNIQUE (username),
    UNIQUE (email)
);
